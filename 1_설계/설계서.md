## 3. 데이터 구조 설계
(프로그램에 필요한 핵심 변수, 배열을 설계하세요. 필요하면 구조체(struct)를 사용할 수도 있습니다.)
import random

# 1. 문제 변수들
questions = [
    "1) 지구의 위성은?",
    "2) 파이썬의 파일 확장자는?"
]

options = [
    ["1. 달", "2. 화성", "3. 목성", "4. 없음"],
    ["1. .exe", "2. .py", "3. .js", "4. .class"]
]

answers = [
    1,  # 1번 정답: '달'
    2   # 2번 정답: '.py'
]

# 2. 기타 변수
score = 0
num_questions = 2  # 원하는 문제 수

# 3. 퀴즈 실행
for i in range(num_questions):
    selected_index = random.randint(0, len(questions)-1)

    print(questions[selected_index])
    for opt in options[selected_index]:
        print(opt)

    user_answer = int(input("정답 번호 입력: "))

    if user_answer == answers[selected_index]:
        print("정답!\n")
        score += 1
    else:
        print("오답!\n")

# 4. 결과 출력
print(f"총 점수: {score}/{num_questions}")
