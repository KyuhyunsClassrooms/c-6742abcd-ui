## 1. `main` 함수의 전체 흐름
(프로젝트의 전체 작동 로직을 의사 코드로 작성하세요.)
FUNCTION main()

    # 1. 프로그램 초기 설정
    INITIALIZE questions 리스트
    INITIALIZE options 리스트
    INITIALIZE answers 리스트
    SET score ← 0

    # 2. 출제할 문제 수 설정
    PRINT "출제할 문제 수를 입력하세요:"
    num_questions ← INPUT()

    # 3. 퀴즈 시작 안내
    PRINT "퀴즈를 시작합니다."

    # 4. 문제 반복 출제
    FOR i FROM 1 TO num_questions DO

        # 4-1. 임의(random) 인덱스 선택
        selected_index ← RANDOM(0 ~ questions 길이 - 1)

        # 4-2. 문제 출력
        PRINT questions[selected_index]

        # 4-3. 보기 출력
        FOR each option IN options[selected_index] DO
            PRINT option
        END FOR

        # 4-4. 사용자 입력 받기
        PRINT "정답 번호를 입력하세요:"
        user_answer ← INPUT()

        # 4-5. 정답 비교
        IF user_answer = answers[selected_index] THEN
            PRINT "정답입니다!"
            score ← score + 1
        ELSE
            PRINT "오답입니다."
        END IF

    END FOR

    # 5. 퀴즈 종료 및 결과 출력
    PRINT "퀴즈 종료!"
    PRINT "총 점수: " + score + "/" + num_questions

END FUNCTION
